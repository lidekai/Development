![](./image/02.01-1.png)

> 写代码

- test.go

```go
package main

import "github.com/gin-gonic/gin"

func main() {
	r := gin.Default() 
	r.GET("/test/*action", func(c *gin.Context) {
		c.String(200,"hello world")
	})
	r.Run() // 启动并监听8080端口
}
```
