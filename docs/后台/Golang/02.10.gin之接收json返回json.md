## 写代码 <a name="go-01" href="#" >:house:</a>

```go
package main

import (
	"net/http"
	"github.com/gin-gonic/gin"
)

type Test struct {
	ID int
	Name string 
	Age int
	Birthday string
}

func main() {
	r := gin.Default() 
	r.GET("/test", testing)
	r.Run() // 启动并监听8080端口
}

func testing(c *gin.Context){
	var test Test
	err :=c.BindJSON(&test)
	if err==nil{
		c.JSON(200, gin.H{
			"id": test.ID,
			"name": test.Name,
			"age": test.Age,
			"birthday": test.Birthday,
		})
	}else{
		c.String(http.StatusOK,"%v","test bind error:%v",err)
	}
}
```
